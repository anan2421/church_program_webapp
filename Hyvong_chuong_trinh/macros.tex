\begin{comment}
\newlength{\subprogwidth}
\newcommand{\subprogram}[3]{%
  \settowidth{\subprogwidth}{\noindent#1\quad\hfill#2\quad\hfill#3}%
  \ifdim\subprogwidth>\textwidth
    % Too wide → shrink to fit
    \resizebox{\textwidth}{!}{\noindent #1\quad\hfill#2\quad\hfill#3}%
  \else
    % Fits → print normally without scaling
    \noindent #1\quad\hfill #2\quad\hfill#3%
  \fi
  \\%
}
\end{comment}
\newlength{\subprogwidth}
\newcommand{\subprogram}[3]{%
  % Create 3 blocks: left, middle, right.
  % Mid block (#2) can contain multiple lines.
  \settowidth{\subprogwidth}{%
    %\begin{tabular}[t]{@{}l@{}}#1\end{tabular}%
    #1
    \quad\hfill
    \begin{tabular}[t]{@{}c@{}}#2\end{tabular}%
    \quad\hfill
    #3
  }%
  \ifdim\subprogwidth>\textwidth
    % Shrink to fit
    \resizebox{\textwidth}{!}{%
    #1
      \quad\hfill
        \begin{tabular}[t]{@{}c@{}}#2\end{tabular}%
      \quad\hfill
      #3
    }%
  \else
    % Normal size
    #1
    \quad\hfill
    \begin{tabular}[t]{@{}c@{}}#2\end{tabular}%
    \quad\hfill
    #3
  \fi
  \\[0.5em]
}
